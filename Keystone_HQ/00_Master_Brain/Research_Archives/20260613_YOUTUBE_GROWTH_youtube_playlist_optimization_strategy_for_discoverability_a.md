def create_new_playlist(youtube, title, description, is_podcast=False):
    body = {
        "snippet": {
            "title": title,
            "description": description
        },
        "status": {
            "privacyStatus": "public"
        }
    }
    
    # Conditionally tag the playlist as a podcast for YouTube Music syndication
    if is_podcast:
        body["status"] = "enabled"

    request = youtube.playlists().insert(
        part="snippet,status",
        body=body
    )
